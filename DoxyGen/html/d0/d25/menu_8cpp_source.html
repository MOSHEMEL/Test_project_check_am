74 void dump_mem()
75 {
76  uint8_t dump_c[40];
77  for (int i = 0; i < APP_SIZE; i++)
78  {
79  if (i % 16 == 0 && i > 0)
80  {

85  sprintf((char*)dump_c, " 0x%02x", *(volatile uint8_t*)(APP_ADDRESS + i));
