44 {

228 {
229  uint32_t maxi = max_pulses;
230  uint32_t percent_remaining = (((maxi - pulses) * 100) / maxi);
231  uint32_t remainder_scaled = (percent_remaining * 11) / 10;
232  /*
233  * Management wanted some buffer to have Graphics show 100% until they use up some amount of pulses
234  * Otherwise after 1 pulse it did ceil(99.9%) and showed 99%, which was not to their liking.
235  * So the function stretches 100% to 110% and then clips it at 100% thus giving ~9% until you
236  * see 99%
237  */
238  if (remainder_scaled > 100)
239  {
240  remainder_scaled = 100;
241  }
