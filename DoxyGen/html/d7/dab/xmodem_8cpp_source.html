29  * @brief This function is the base of the Xmodem protocol.
30  * When we receive a header from UART, it decides what action it shall take.
31  * @param void
32  * @return void
33  */
34 void xmodem_receive(void)
35 {
36  volatile xmodem_status status = X_OK;
37  uint8_t error_number = 0u;
38 
39  x_first_packet_received = false;
40  xmodem_packet_number = 1u;
41  xmodem_actual_flash_address = APPLICATION_ADDRESS;
42 
43  /* Loop until there isn't any error (or until we jump to the user application). */
44  while (X_OK == status)
45  {
46  uint8_t header = 0x00u;
47 
48  /* Get the header from UART. */
49  HAL_StatusTypeDef comm_status = HAL_UART_Receive(SERIAL_PC, &header, 1u, 100);
50 
51  /* Spam the host (until we receive something) with ACSII "C", to notify it, we want to use CRC-16. */
52  if ((HAL_OK != comm_status) && (false == x_first_packet_received))
53  {
54  (void)Serial_PutByte(X_C);
55  }
56  /* Uart timeout or any other errors. */
57  else if ((HAL_OK != comm_status) && (true == x_first_packet_received))
58  {
59  status = xmodem_error_handler(&error_number, X_MAX_ERRORS);
60  }
61  else
62  {
63  /* Do nothing. */
64  }
65 
66  xmodem_status packet_status = X_ERROR;
67  packet_status = xmodem_handle_packet(header);
68  /* The header can be: SOH, STX, EOT and CAN. */
69 
70  switch (header)
71  {
72  /* 128 or 1024 bytes of data. */
73  case X_SOH:
74  /* If the handling was successful, then send an ACK. */
75  if (X_OK == packet_status)
76  {
77  (void)Serial_PutByte(X_ACK);
78  }
79  /* If the error was flash related, then immediately set the error counter to max (graceful abort). */
80  else if(X_ERROR_FLASH == packet_status)
81  {
82  error_number = X_MAX_ERRORS;
83  status = xmodem_error_handler(&error_number, X_MAX_ERRORS);
84  }
85  /* Error while processing the packet, either send a NAK or do graceful abort. */
86  else
87  {
88  status = xmodem_error_handler(&error_number, X_MAX_ERRORS);
89  }
90  break;
91  case X_STX:
92  /* If the handling was successful, then send an ACK. */
93  if (X_OK == packet_status)
94  {
95  (void)Serial_PutByte(X_ACK);
96  }
97  /* If the error was flash related, then immediately set the error counter to max (graceful abort). */
98  else if (X_ERROR_FLASH == packet_status)
99  {
100  error_number = X_MAX_ERRORS;
101  status = xmodem_error_handler(&error_number, X_MAX_ERRORS);
102  }
103  /* Error while processing the packet, either send a NAK or do graceful abort. */
104  else
105  {
106  status = xmodem_error_handler(&error_number, X_MAX_ERRORS);
107  }
108  break;
109  /* End of Transmission. */
110  case X_EOT:
111  /* ACK, feedback to user (as a text), then jump to user application. */
112  Serial_PutByte(X_ACK);
113  Serial_PutString("\n\rFirmware updated!\n\r");
114 
115  // Write out size of app
116  memory_extern->write(1, 0xfff04, xmodem_actual_flash_address);
117  uint8_t size_c[30];
118  sprintf((char*)size_c, "Write Out: %d\r\n", xmodem_actual_flash_address);
119  HAL_UART_Transmit(SERIAL_PC, size_c, strlen((char*)size_c), 10);
120  // Write out version of app
121  uint8_t ver[30];
122  sprintf((char*)ver, "VERSION: %s\r\n", VERSION);
123  HAL_UART_Transmit(SERIAL_PC, ver, strlen((char*)ver), 10);
124 
125  memory_extern->write(1, 0xfff08, MCU_VERSION_MAJOR);
126  memory_extern->write(1, 0xfff0c, MCU_VERSION_MINOR);
127  memory_extern->write(1, 0xfff10, MCU_VERSION_PATCH);
128  memory_extern->write(1, 0xfff14, MCU_VERSION_RC);
129 
130  Serial_PutString("Jumping to user application...\n\r");
131  HAL_Delay(2500);
132  NVIC_SystemReset();
133  break;
134  /* Abort from host. */
135  case X_CAN:
136  status = X_ERROR;
137  break;
138  default:
139  /* Wrong header. */
140  if (HAL_OK == comm_status)
141  {
142  status = xmodem_error_handler(&error_number, X_MAX_ERRORS);
143  }
144  break;
145  }
146  }

149 /**
150  * @brief Calculates the CRC-16 for the input package.
151  * @param *data: Array of the data which we want to calculate.
152  * @param length: Size of the data, either 128 or 1024 bytes.
153  * @return status: The calculated CRC.
154  */
155 uint16_t xmodem_calc_crc(uint8_t* data, uint16_t length)
156 {
157  uint16_t crc = 0u;
158  while (length)
159  {
160  length--;
161  crc = crc ^ ((uint16_t)*data++ << 8u);
162  for (uint8_t i = 0u; i < 8u; i++)
163  {
164  if (crc & 0x8000u)
165  {
166  crc = (crc << 1u) ^ 0x1021u;
167  }
168  else
169  {
170  crc = crc << 1u;
171  }
172  }
173  }
174  return crc;
175 }

177 /**
178  * @brief This function handles the data packet we get from the xmodem protocol.
179  * @param header: SOH or STX.
180  * @return status: Report about the packet.
181  */
182 static xmodem_status xmodem_handle_packet(uint8_t header)
183 {
184  xmodem_status status = X_OK;
185  uint16_t size = 0u;
186  if (X_SOH == header)
187  {
188  size = X_PACKET_128_SIZE;
189  }
190  else if (X_STX == header)
191  {
192  size = X_PACKET_1024_SIZE;
193  }
194  else if (0x00 == header || 0xff == header)
195  {
196  return X_OK;
197  }
198  else
199  {
200  /* Wrong header type. */
201  status = X_ERROR;
202  }
203  uint16_t length = size + X_PACKET_DATA_INDEX + X_PACKET_CRC_SIZE;
204 
205  /* Get the packet (except for the header) from UART. */
206  HAL_StatusTypeDef comm_status = HAL_OK;
207  for (unsigned int j = 0; j < length; j++)
208  {
209  comm_status = HAL_UART_Receive(SERIAL_PC, &aPacketData[j], 1u, 1000);
210  }
211  /*
212  for (unsigned int j = 0; j < length; j++)
213  {
214  HAL_UART_Transmit(SERIAL_PC, &aPacketData[j], 1u, 999);
215  }
216  */
217  /* The last two bytes are the CRC from the host. */
218  uint16_t crc_received = ((uint16_t)aPacketData[size + 2u] << 8u) | ((uint16_t)aPacketData[size + 3u]);
219  /* We calculate it too. */
220  uint16_t crc_calculated = xmodem_calc_crc(&aPacketData[X_PACKET_DATA_INDEX], size);
221 
222 
223  /* Error handling and flashing. */
224  if (X_OK == status)
225  {
226  if (HAL_OK != comm_status)
227  {
228  /* UART error. */
229  status = X_ERROR_UART;
230  }
231  if (xmodem_packet_number != aPacketData[X_PACKET_NUMBER_INDEX])
232  {
233  /* Packet number counter mismatch. */
234  status = X_ERROR_NUMBER;
235  }
236  if (255u != (aPacketData[X_PACKET_NUMBER_INDEX] + aPacketData[X_PACKET_NUMBER_COMPLEMENT_INDEX]))
237  {
238  /* The sum of the packet number and packet number complement aren't 255. */
239  /* The sum always has to be 255. */
240  status = X_ERROR_NUMBER;
241  }
242  if (crc_calculated != crc_received)
243  {
244  /* The calculated and received CRC are different. */
245  status = X_ERROR_CRC;
246  }
247  /* Do the actual flashing. */
248  if (HAL_OK != FLASH_If_Write(xmodem_actual_flash_address, &aPacketData[2],(uint32_t)size / 4u))
249  {
250  /* Flashing error. */
251  status = X_ERROR_FLASH;
252  }
253  }
254 
255  /* Raise the packet number and the address counters (if there wasn't any error). */
256  if (X_OK == status)
257  {
258  xmodem_packet_number++;
259  xmodem_actual_flash_address += size;
260  }
261 
262  return status;
263 }

265 /**
266  * @brief Handles the xmodem error.
267  * Raises the error counter, then if the number of the errors reached critical, do a graceful abort, otherwise send a NAK.
268  * @param *error_number: Number of current errors (passed as a pointer).
269  * @param max_error_number: Maximal allowed number of errors.
270  * @return status: X_ERROR in case of too many errors, X_OK otherwise.
271  */
272 static xmodem_status xmodem_error_handler(uint8_t* error_number, uint8_t max_error_number)
273 {
274  xmodem_status status = X_OK;
275  /* Raise the error counter. */
276  (*error_number)++;
277  /* If the counter reached the max value, then abort. */
278  if ((*error_number) >= max_error_number)
279  {
280  /* Graceful abort. */
281  (void)Serial_PutByte(X_CAN);
282  (void)Serial_PutByte(X_CAN);
283  status = X_ERROR;
284  }
285  /* Otherwise send a NAK for a repeat. */
286  else
287  {
288  (void)Serial_PutByte(X_NAK);
289  status = X_OK;
290  }
291  return status;
292 }
